console.log(index_indexOf !== -1);      // true 
console.log(arr.includes(65))           // true

console.log([NaN].indexOf(NaN) !== -1); // false

以上，在面试中，只能得到70分，下面是stackoverflow给的更完整的版：

var deepEqual = function (x, y) {
  if (x === y) {
    return true;
  }
  // 如果使用 typeof 去判断 x 是否为复杂对象，
  // 固定写法就是：1.判断是 object 2.不是 null
  else if ((typeof x == "object" && x != null) && (typeof y == "object" && y != null)) {
    if (Object.keys(x).length != Object.keys(y).length)
      return false;
		// 遍历 key 值有两种方式
    for (var prop in x) { 
      // + 多考虑了原型链
      if (y.hasOwnProperty(prop))
      {  
        if (!deepEqual(x[prop], y[prop]))
          return false;
      }
      else
        return false;
    }
    return true;
  }
  else 
    return false;
}

// shallowEqual（数组版）：使用 every + includes 的方式来保证
const a = [1, 2, 3];
const b = [4, 5, 6];
const c = [3, 2, 1];

function shallowEqual(a, b) {
    return Array.isArray(a) &&
        Array.isArray(b) &&
        a.length === b.length &&
        a.every((val) => b.includes(val));
}
